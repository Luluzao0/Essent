rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Permitir leitura e escrita para usuários autenticados
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Regras específicas para coleções da aplicação
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /wellness_data/{document} {
      allow read, write: if request.auth != null;
    }
    
    match /user_progress/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /daily_goals/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /meditation_sessions/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}

INSTRUÇÕES RÁPIDAS:
1. Abra console.firebase.google.com
2. Selecione seu projeto
3. Vá em "Firestore Database" > "Rules"
4. Cole as regras acima
5. Clique em "Publicar"
